2. 모든 리스트 가지고 오기 (연습) 

```2=java
@Controller //이 클래스는 컨트롤러인데, pojo구나 
public class BoardController {
	// /boardList.do 요청이 오면 아래쪽의 메서드가 실행된다.
	// HandlerMapping
	@RequestMapping("/boardList.do") 
	public String boardList(Model model) {
		//기존에는 ServletRequest 객체였는데 스프링은 Model의 객체로 객체바인딩 가능
		Board vo=new Board();
		vo.setIdx(1);
		vo.setTitle("게시판실습");
		vo.setContent("게시판실습");
		vo.setWriter("태연");
		vo.setIndate("2023-06-24");
		vo.setCount(0);
		
		List<Board> list=new ArrayList<>();
		list.add(vo);
		list.add(vo);
		list.add(vo);
		
		model.addAttribute("list", list); //객체 바인딩 
		//boardList.jsp에서는 EL로 받으면 된다.
		
		return "boardList"; // /WEB-INF/views/boardList.jsp
		//ViewResolver가 내부적으로 제공된다. -> forwarding
	}
}
```
